function initTimer() {
    const now = Date.now();
    let endTime = localStorage.getItem('aitrainer_endTime');
    if (!endTime || now > parseInt(endTime)) {
        endTime = now + 24 * 60 * 60 * 1000;
        localStorage.setItem('aitrainer_endTime', endTime);
    }
    endTime = parseInt(endTime);

    function updateTimer() {
        const remaining = Math.max(0, endTime - Date.now());
        if (remaining === 0) {
            endTime = Date.now() + 24 * 60 * 60 * 1000;
            localStorage.setItem('aitrainer_endTime', endTime);
        }
        const hrs = Math.floor(remaining / (1000 * 60 * 60));
        const mins = Math.floor((remaining % (1000 * 60 * 60)) / (1000 * 60));
        const secs = Math.floor((remaining % (1000 * 60)) / 1000);
        const str = String(hrs).padStart(2,'0') + ':' + String(mins).padStart(2,'0') + ':' + String(secs).padStart(2,'0');
        document.getElementById('topTimer').textContent = str;
        document.getElementById('finalTimer').textContent = str;
    }
    updateTimer();
    setInterval(updateTimer, 1000);
}

function initSlots() {
    let slots = localStorage.getItem('aitrainer_slots');
    let lastUpdate = localStorage.getItem('aitrainer_slotsLastUpdate');
    const now = Date.now();
    if (!slots || !lastUpdate) {
        slots = 10;
        lastUpdate = now;
        localStorage.setItem('aitrainer_slots', slots);
        localStorage.setItem('aitrainer_slotsLastUpdate', lastUpdate);
    } else {
        slots = parseInt(slots);
        lastUpdate = parseInt(lastUpdate);
        const elapsed = Math.floor((now - lastUpdate) / 1000);
        const drops = Math.floor(elapsed / 45);
        slots = Math.max(1, slots - drops);
        if (slots <= 1) slots = 10;
        localStorage.setItem('aitrainer_slots', slots);
        localStorage.setItem('aitrainer_slotsLastUpdate', now);
    }

    function updateSlotsDisplay() {
        document.getElementById('topSlots').textContent = slots;
        document.getElementById('finalSlots').textContent = slots;
        document.getElementById('stickySlots').textContent = slots;
    }
    updateSlotsDisplay();

    setInterval(function() {
        slots--;
        if (slots < 1) slots = 10;
        localStorage.setItem('aitrainer_slots', slots);
        localStorage.setItem('aitrainer_slotsLastUpdate', Date.now());
        updateSlotsDisplay();
    }, 45000);
}

function initStickyCta() {
    var sticky = document.getElementById('stickyCta');
    var footer = document.getElementById('footer');
    var shown = false;
    window.addEventListener('scroll', function() {
        var scrollPercent = window.scrollY / (document.body.scrollHeight - window.innerHeight);
        var footerTop = footer.getBoundingClientRect().top;
        var nearFooter = footerTop < window.innerHeight + 100;
        if (scrollPercent > 0.35 && !nearFooter && !shown) {
            sticky.classList.add('visible');
            shown = true;
        } else if ((scrollPercent <= 0.35 || nearFooter) && shown) {
            sticky.classList.remove('visible');
            shown = false;
        }
    });
}

function showExitPopup() {

}

function hideExitPopup() {
    exitOverlay.classList.remove('active');
    document.body.style.overflow = '';
}

    if (e.clientY < 10 && !exitShown) {
        showExitPopup();
    }
});

// Close on overlay click

document.addEventListener('keydown', function(e) {
    if (e.key === 'Escape' && exitOverlay.classList.contains('active')) {
        hideExitPopup();
    }
